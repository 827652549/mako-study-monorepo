核心原理：手动解析 SSE 帧
// SSE 协议：多行构成一帧，空行分割帧
// 例如一帧原始文本：
//   "event: message_text\ndata: {\"text\":\"好的，\"}\n\n"

function parseSseChunk(rawText) {
  const frames = rawText.split('\n\n').filter(Boolean);
  return frames.map(frame => {
    const lines = frame.split('\n');
    const result = {};
    lines.forEach(line => {
      if (line.startsWith('event:')) result.event = line.slice(6).trim();
      if (line.startsWith('data:'))  result.data  = line.slice(5).trim();
    });
    return result;
  });
}
